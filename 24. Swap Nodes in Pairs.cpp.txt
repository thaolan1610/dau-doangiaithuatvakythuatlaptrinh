class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode* dummy = new ListNode(0);
        dummy->next = head;

        ListNode* prev = dummy;

        while (prev->next && prev->next->next) {
            ListNode* first = prev->next;
            ListNode* second = first->next;

            // swap
            first->next = second->next;
            second->next = first;
            prev->next = second;

            // move prev
            prev = first;
        }

        return dummy->next;
    }
};

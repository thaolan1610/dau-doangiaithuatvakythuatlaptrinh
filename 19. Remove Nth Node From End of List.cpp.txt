class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* dummy = new ListNode(0);
        dummy->next = head;

        ListNode* fast = dummy;
        ListNode* slow = dummy;

        // fast đi trước n + 1 bước
        for (int i = 0; i <= n; i++) {
            fast = fast->next;
        }

        // di chuyển cả 2 con trỏ
        while (fast != nullptr) {
            fast = fast->next;
            slow = slow->next;
        }

        // xoá node thứ n từ cuối
        slow->next = slow->next->next;

        return dummy->next;
    }
};
